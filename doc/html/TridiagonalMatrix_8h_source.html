215  assert(size == vec.size());
216  Vector<T> d(vec);
217  auto c = mat[Lower];
218  auto b = mat[Diagonal];
219  auto a = mat[Upper];

222  c[0] /= c[0];
223  d[0] /= b[0];
224  for (unsigned int i = 1; i < size - 1; ++i) {
225  c[i] /= b[i] - a[i] * c[i - 1];
226  d[i] = (d[i] - a[i] * d[i - 1]) / (b[i] - a[i] * c[i - 1]);

334  std::get<0>(result) = sign(a); //std::copysign(1.0, a); // c

337  } else if (std::abs(a) <= std::abs(std::numeric_limits<T>::epsilon())) {

357  void qrStep(unsigned int iterations) {
358  unsigned int m = (*this).getSize() - 1;
359  unsigned int k = iterations;
360  while (m > 1) {
361  T shift = (*this)(Diagonal, m);
362  const T u = ((*this)(Diagonal, m - 1) - (*this)(Diagonal, m)) / 2.0;
363  if (std::abs(u) <= std::abs(std::numeric_limits<T>::epsilon())) {
364  shift = (*this)(Diagonal, m) - std::abs((*this)(Upper, m));
365  } else {
366  T b = (*this)(Upper, m - 1);
367  shift = (*this)(Diagonal, m) - sign(u) * b * b / (std::abs(u) + std::sqrt(u * u + b * b));
368  }
369 
370  T x = (*this)(Diagonal, 0) - shift;
371  T y = (*this)(Upper, 0);
372  for (unsigned int k = 0; k < m; ++k) {
373  std::tuple<T, T, T> giv = givens(x, y);
374  const T c = std::get<0>(giv);
375  const T s = std::get<1>(giv);
376 
377  const T w = c * x - s * y;
378  const T d = (*this)(Diagonal, k) - (*this)(Diagonal, k + 1);
379  const T z = (2 * c * (*this)(Upper, k) + d * s) * s;
380  (*this)(Diagonal, k) -= z;
381  (*this)(Diagonal, k + 1) += z;
382  (*this)(Upper, k) = d * c * s + (c * c - s * s) * (*this)(Upper, k);
383  x = (*this)(Upper, k);
384  if (k > 0) {
385  (*this)(Upper, k - 1) = w;
386  }
387  if (k < m - 1) {
388  y = -s * (*this)(Upper, k + 1);
389  (*this)(Upper, k + 1) *= c;
390  }
391  }
392 
393  k--;
394  if (k == 0) {
395  m--;
396  k = iterations;
397  }
398  }
